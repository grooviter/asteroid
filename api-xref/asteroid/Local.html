1   package asteroid;
2   
3   import org.codehaus.groovy.transform.GroovyASTTransformationClass;
4   
5   import java.lang.annotation.ElementType;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;
8   import java.lang.annotation.Target;
9   
10  /**
11   * Marks a given annotation as a <b>local</b> AST transformation
12   * marker
13   * <br>
14   *
15   * <pre><code>
16   * {@literal @}Local(MyTransformationImplementation)
17   * {@literal @}interface MyAnnotation {
18   *     //...
19   * }
20   * </code></pre>
21   * Then you can use your annotation:
22   * <pre><code>
23   * {@literal @}MyAnnotation
24   *  class SomeType {
25   *     //...
26   *  }
27   * </code></pre>
28   * @since 0.1.0
29   */
30  @Target(ElementType.ANNOTATION_TYPE)
31  @Retention(RetentionPolicy.SOURCE)
32  @GroovyASTTransformationClass("asteroid.internal.LocalTransformation")
33  public @interface Local {
34      /**
35       * Sets the transformation implementation class
36       *
37       * @return a class representing an AST transformation implementation
38       * @since 0.1.0
39       */
40      Class<? extends AbstractLocalTransformation> value();
41  
42      /**
43       * @since 0.2.0
44       */
45      TO applyTo() default TO.TYPE;
46  
47      /**
48       * Targets available to apply a specific transformation
49       *
50       * @since 0.1.0
51       */
52      public static enum TO {
53          /**
54           * Applies to a given class field
55           *
56           * @since 0.1.0
57           */
58          FIELD,
59          /**
60           * Applies to a given class
61           *
62           * @since 0.1.0
63           */
64          TYPE,
65          /**
66           * Applies to a given method
67           *
68           * @since 0.1.0
69           */
70          METHOD,
71          /**
72           * Applies to a {@link org.codehaus.groovy.ast.ConstructorNode}
73           *
74           * @since 0.2.5
75           */
76          CONSTRUCTOR,
77          /**
78           * Applies to a {@link org.codehaus.groovy.ast.Variable}
79           *
80           * @since 0.2.5
81           */
82          LOCAL_VARIABLE,
83          /**
84           * Applies to a {@link org.codehaus.groovy.ast.PackageNode}
85           *
86           * @since 0.2.5
87           */
88          PACKAGE,
89          /**
90           * Applies to a {@link org.codehaus.groovy.ast.Parameter}
91           *
92           * @since 0.2.5
93           */
94          PARAMETER,
95          /**
96           * Applies to any {@link org.codehaus.groovy.ast.AnnotatedNode}
97           *
98           * @since 0.2.5
99           */
100         ANNOTATED
101     }
102 }
