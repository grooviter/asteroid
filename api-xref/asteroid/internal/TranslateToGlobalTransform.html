1   package asteroid.internal;
2   
3   import org.codehaus.groovy.control.SourceUnit;
4   import org.codehaus.groovy.control.CompilePhase;
5   import org.codehaus.groovy.ast.ClassNode;
6   
7   import org.codehaus.groovy.GroovyBugError;
8   import org.codehaus.groovy.ast.AnnotationNode;
9   
10  import asteroid.A;
11  import asteroid.Phase;
12  import asteroid.AbstractGlobalTransformation;
13  import asteroid.transformer.AbstractClassNodeTransformer;
14  
15  /**
16   * This {@link AbstractClassNodeTransformer} will be applied to all {@link
17   * ClassNode} instances annotated with {@link Phase}.
18   * <br><br>
19   * It will add certain low level annotations to trigger the process of
20   * the annotated transformation. These annotations are the java
21   * retention and target annotations, and the Groovy transform
22   * annotation.
23   *
24   * @since 0.1.6
25   */
26  public class TranslateToGlobalTransform extends AbstractClassNodeTransformer {
27  
28      private static final String BLANK = "";
29      private static final String TX_NAME = "Phase";
30      private static final String PHASE_PFIX = "Phase.GLOBAL.";
31      private static final String PHASE_PFIX_SHORT = "GLOBAL.";
32      private static final String PHASE_GROOVY = "CompilePhase.";
33      private static final String PHASE_WRONG = "GlobalAnnotation compilation phase is wrong!!";
34      private static final String PHASE_MISSING = "GlobalAnnotation compilation phase is missing!!";
35  
36      /**
37       * Constructor receiving the {@link SourceUnit}
38       *
39       * @param sourceUnit
40       * @since 0.1.6
41       */
42      public TranslateToGlobalTransform(final SourceUnit sourceUnit) {
43          super(sourceUnit, A.CRITERIA.byAnnotationSimpleName(TX_NAME));
44      }
45  
46      /**
47       * {@inheritDoc}
48       */
49      @Override
50      public void transformClass(final ClassNode annotated) {
51          final ClassNode superClass = annotated.getSuperClass();
52  
53          if (superClass == null) {
54              return;
55          }
56  
57          final ClassNode reference = A.NODES.clazz(AbstractGlobalTransformation.class).build();
58          final Boolean isGlobal = A.UTIL.NODE.isOrExtendsUnsafe(annotated, reference);
59  
60          if (!isGlobal) {
61              return;
62          }
63  
64          final AnnotationNode annotation = A.UTIL.NODE.getAnnotationFrom(annotated, TX_NAME);
65          final CompilePhase phase = extractCompilePhaseFromSafely(annotation);
66  
67          TransformationUtils.addASTAnnotationsFromTo(annotated, phase);
68          A.UTIL.NODE.removeAnnotation(annotated, annotation);
69      }
70  
71      @SuppressWarnings("PMD.AvoidCatchingGenericException")
72      private CompilePhase extractCompilePhaseFromSafely(final AnnotationNode annotationNode) {
73          try {
74              return extractCompilePhaseFrom(annotationNode);
75          } catch(IllegalArgumentException iaex){
76              throw new GroovyBugError(PHASE_WRONG, iaex);
77          } catch(Exception ex) {
78              throw new GroovyBugError(PHASE_MISSING, ex);
79          }
80      }
81  
82      private CompilePhase extractCompilePhaseFrom(final AnnotationNode annotationNode) {
83          final String       value         = A.UTIL.NODE.getStringValue(annotationNode);
84          final String       phaseAsString = value
85              .replace(PHASE_GROOVY, BLANK)
86              .replace(PHASE_PFIX, BLANK)
87              .replace(PHASE_PFIX_SHORT, BLANK);
88          final CompilePhase compilePhase  = CompilePhase.valueOf(phaseAsString);
89  
90          return compilePhase;
91      }
92  }
