9   import groovy.lang.Closure;
10  import org.codehaus.groovy.ast.ASTNode;
11  import org.codehaus.groovy.ast.AnnotatedNode;

24  import java.lang.annotation.ElementType;
25  import java.lang.annotation.Retention;

28  import java.util.Arrays;

31  /**
32   * Reduces the boiler plate code when declaring an annotation as possible target for a given
33   * transformation.
34   *
35   * It adds code such as the {@link java.lang.annotation.Retention} and {@link java.lang.annotation.Target} annotations as well as the
36   * {@link org.codehaus.groovy.transform.GroovyASTTransformationClass}.
37   *
38   * This code:
39   *
40   * <pre>
41   * <code>
42   *   import my.transformation.MyTransformation
43   *
44   *   {@literal @}Local(MyTransformation)
45   *   public @interface MyAnnotation { }
46   * </code>
47   * </pre>
48   *
49   * Will become:
50   *
51   * <pre>
52   * <code>
53   *    {@literal @}Retention(RetentionPolicy.SOURCE)
54   *    {@literal @}Target([ElementType.METHOD])
55   *    {@literal @}GroovyASTTransformation("my.transformation.MyTransformation")
56   *    public @interface MyAnnotation { }
57   * </code>
58   * </pre>
59   *
60   * @since 0.1.0
61   *
62   */
63  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
64  public class LocalTransformation extends AbstractASTTransformation {
65  
66      /**
67       * {@inheritDoc}
68       *
69       * @since 0.1.0
70       */
71      @Override
72      public void visit(final ASTNode[] nodes, final SourceUnit source) {
73          this.sourceUnit = source;
74  
75          if (!check(nodes)) {
76              return;
77          }
78  
79          final AnnotationNode annotationNode = A.UTIL.NODE.getFirstNodeAs(nodes, AnnotationNode.class);
80          final ClassNode      annotatedNode  = A.UTIL.NODE.getLastNodeAs(nodes, ClassNode.class);
81  
82          addAnnotationsFromTo(annotationNode, annotatedNode);
83      }
84  
85      private void addAnnotationsFromTo(final AnnotationNode annotationNode, final ClassNode annotatedNode) {
86          final String qualifiedName = A.UTIL.NODE.get(annotationNode, String.class);
87          final String target        = A.UTIL.NODE.get(annotationNode, "applyTo", String.class);
88  
89          annotatedNode.addAnnotation(getTargetAnnotation(target));
90          annotatedNode.addAnnotation(getRetentionAnnotation());
91          annotatedNode.addAnnotation(getGroovyAnnotationWith(qualifiedName));
92      }
93  
94      private AnnotationNode getTargetAnnotation(final String resolvedTarget) {
95          final List<ElementType> types = resolveAnnotationTarget(resolvedTarget);
96          final ListExpression listExpr = resolveTargetFromElementType(types);
97  
98          return A.NODES.annotation(Target.class)
99                  .member(A.UTIL.NODE.ANNOTATION_VALUE, listExpr)
100                 .build();
101     }
102 
103     private ListExpression resolveTargetFromElementType(final List<ElementType> types) {
104         final List<PropertyExpression> expressions =
105                 collect(types, new Closure<PropertyExpression>(null) {
106                     PropertyExpression doCall(final ElementType type) {
107                         return A.EXPR.propX(
108                                 A.EXPR.classX(ElementType.class),
109                                 A.EXPR.constX(type.toString()));
110                     }
111                 });
112 
113         return A.EXPR.listX(expressions.toArray(new PropertyExpression[expressions.size()]));
114 
115     }
116 
117     private List<ElementType> resolveAnnotationTarget(final String target) {
118         if (target == null) {
119             return Arrays.asList(ElementType.TYPE);
120         }
121 
122         if ("ANNOTATED".equals(target)) {
123             return Arrays
124                     .asList(ElementType.ANNOTATION_TYPE,
125                             ElementType.CONSTRUCTOR,
126                             ElementType.FIELD,
127                             ElementType.LOCAL_VARIABLE,
128                             ElementType.METHOD,
129                             ElementType.PACKAGE,
130                             ElementType.PARAMETER,
131                             ElementType.TYPE);
132         }
133 
134         return Arrays.asList(ElementType.valueOf(target));
135     }
136 
137     private AnnotationNode getRetentionAnnotation() {
138         final ClassExpression    classExpr    = A.EXPR.classX(RetentionPolicy.class);
139         final ConstantExpression constantExpr = A.EXPR.constX(RetentionPolicy.SOURCE.toString());
140         final PropertyExpression propertyExpr = A.EXPR.propX(classExpr, constantExpr);
141 
142         return A.NODES.annotation(Retention.class)
143                 .member(A.UTIL.NODE.ANNOTATION_VALUE, propertyExpr)
144                 .build();
145     }
146 
147     private AnnotationNode getGroovyAnnotationWith(final String qualifiedName) {
148         final ConstantExpression constant = A.EXPR.constX(qualifiedName);
149 
150         return A.NODES.annotation(GroovyASTTransformationClass.class)
151                 .member(A.UTIL.NODE.ANNOTATION_VALUE, constant)
152                 .build();
153     }
154 
155     private boolean check(final ASTNode... nodes) {
156         return !(thereIsNoNodes(nodes)                               ||
157                 thereAreOtherThanTwo(nodes)                          ||
158                 firstNodeIsNotAnAnnotation(nodes)                    ||
159                 firstNodeIsNotAnAnnotationOfType(nodes, Local.class) ||
160                 lastNodeIsNotAnAnnotatedNode(nodes));
161     }
162 
163     private boolean thereIsNoNodes(final ASTNode... nodes) {
164         return nodes == null;
165     }
166 
167     private boolean thereAreOtherThanTwo(final ASTNode... nodes) {
168         return nodes.length != 2;
169     }
170 
171     private boolean firstNodeIsNotAnAnnotation(final ASTNode... nodes) {
172         return !(first(nodes) instanceof AnnotationNode);
173     }
174 
175     private boolean lastNodeIsNotAnAnnotatedNode(final ASTNode... nodes) {
176         return !(last(nodes) instanceof AnnotatedNode);
177     }
178 
179     private boolean firstNodeIsNotAnAnnotationOfType(final ASTNode[] nodes, final Class annotationType) {
180         final AnnotationNode annotation = A.UTIL.NODE.getFirstNodeAs(nodes, AnnotationNode.class);
181         final ClassNode annotationClass = A.NODES.clazz(annotationType).build();
182 
183         return !annotation.getClassNode().isDerivedFrom(annotationClass);
184     }
