1   package asteroid.nodes;
2   
3   import org.codehaus.groovy.ast.ClassNode;
4   import org.codehaus.groovy.ast.Parameter;
5   
6   /**
7    * Creates a new {@link Parameter} instance
8    *
9    * @since 0.1.0
10   */
11  @SuppressWarnings("PMD.AvoidFieldNameMatchingMethodName")
12  public class ParameterNodeBuilder {
13  
14      private final String name;
15      private ClassNode type;
16  
17      /**
18       * Default constructor
19       *
20       * @param name of the parameter
21       */
22      public ParameterNodeBuilder(final String name) {
23          this.name = name;
24      }
25  
26      /**
27       * Creates a new {@link ParameterNodeBuilder}
28       *
29       * @param name of the parameter
30       * @return a new instance of {@link ParameterNodeBuilder}
31       */
32      public static ParameterNodeBuilder param(final String name) {
33          return new ParameterNodeBuilder(name);
34      }
35  
36      /**
37       * Sets the type of the parameter
38       *
39       * @param type the {@link ClassNode} of the
40       */
41      public ParameterNodeBuilder type(final ClassNode type) {
42          this.type = type;
43          return this;
44      }
45  
46      /**
47       * Returns the {@link Parameter} instance
48       *
49       * @return an instance of {@link Parameter}
50       */
51      public Parameter build() {
52          return new Parameter(type, name);
53      }
54  }
