1   package asteroid.nodes;
2   
3   import org.codehaus.groovy.ast.ClassNode;
4   import org.codehaus.groovy.ast.GenericsType;
5   
6   /**
7    * Builder to create instances of type {@link GenericsType}
8    *
9    * @since 0.1.0
10   */
11  final public class GenericsTypeBuilder {
12  
13      private final GenericsType genericsType;
14  
15      /**
16       * Sets the type of the generic type
17       *
18       * @param classNode Sets the {@link ClassNode} of the generic type
19       * @since 0.1.0
20       */
21      private GenericsTypeBuilder(final ClassNode classNode) {
22          this.genericsType = new GenericsType(classNode);
23      }
24  
25      /**
26       * Sets the type of the generic type
27       *
28       * @param classNode Sets the {@link ClassNode} of the generic type
29       * @return an instance of type {@link GenericsTypeBuilder}
30       * @since 0.1.0
31       */
32      public static GenericsTypeBuilder generics(final ClassNode classNode) {
33          return new GenericsTypeBuilder(classNode);
34      }
35  
36      /**
37       * Returns the configured generic type
38       *
39       * @return an instance of type {@link GenericsType}
40       * @since 0.1.0
41       */
42      public GenericsType build() {
43          return this.genericsType;
44      }
45  }
