1   package asteroid.transformer;
2   
3   import groovy.lang.Closure;
4   
5   import org.codehaus.groovy.control.SourceUnit;
6   import org.codehaus.groovy.ast.stmt.Statement;
7   
8   /**
9    * This {@link Transformer} could be used as a base for transforming
10   * {@link Statement} instances.
11   * <br><br>
12   * <b class="warning">IMPORTANT:</b> The parameter type is only used as a hint. If you
13   * are not carefull on how you define the search criteria you could
14   * get a {@link ClassCastException} at runtime. The criteria should ask
15   * for the type of the statement in the first place.
16   *
17   * @param <T> use as a hint for the {@link AbstractStatementTransformer#transformStatement} method parameter
18   * @since 0.2.0
19   */
20  public abstract class AbstractStatementTransformer<T extends Statement> extends AbstractTransformer {
21  
22      private final Closure<Boolean> criteria;
23  
24      /**
25       * Every instance needs the source unit awareness and the name of the method
26       * it's going to transform
27       *
28       * @param sourceUnit Needed to apply scope
29       * @param criteria the criteria used to search the interesting
30       * {@link Statement}
31       * @since 0.2.0
32       */
33      public AbstractStatementTransformer(final SourceUnit sourceUnit, final Closure<Boolean> criteria) {
34          super(sourceUnit);
35          this.criteria = criteria;
36      }
37  
38      /**
39       * {@inheritDoc}
40       */
41      @Override
42      public void visitStatement(final Statement statement) {
43          if (criteria.call(statement)) {
44              transformStatement((T) statement);
45              return;
46          }
47          statement.visit(this);
48      }
49  
50      /**
51       * This method will transform the statement into its final version.
52       *
53       * @param statement the method statement you want to transform
54       * @since 0.2.0
55       */
56      public abstract void transformStatement(T statement);
57  }
