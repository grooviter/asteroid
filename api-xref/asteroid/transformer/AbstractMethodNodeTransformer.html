1   package asteroid.transformer;
2   
3   import org.codehaus.groovy.ast.MethodNode;
4   
5   import org.codehaus.groovy.control.SourceUnit;
6   import groovy.lang.Closure;
7   
8   /**
9    * This {@link Transformer} can be used to transform {@link MethodNode}
10   * nodes. To locate the classes you want to transform, a criteria in
11   * form of {@link Closure} can be passed in the constructor.
12   * The criteria must be a {@link Closure} receiving as unique parameter
13   * a {@link MethodNode} and returning a {@link Boolean} value. If the closure
14   * returns true, then the transformation will be applied.
15   * </br></br>
16   * There are some static methods creating some default criterias:
17   * </br></br>
18   * <ul>
19   *    <li>{@link asteroid.Criterias#byMethodNodeNameContains(String)}</li>
20   *    <li>{@link asteroid.Criterias#byMethodNodeNameStartsWith(String)}</li>
21   *    <li>{@link asteroid.Criterias#byMethodNodeNameEndsWith(String)}</li>
22   * </ul>
23   *
24   * @since 0.2.0
25   */
26  public abstract class AbstractMethodNodeTransformer extends AbstractTransformer {
27  
28      private final Closure<Boolean> criteria;
29  
30      /**
31       * Every instance needs the source unit and the name of the class
32       * it's going to transform.
33       *
34       * To find methods eligible to be transformed by this {@link Transformer}
35       * a {@link Closure} returning a boolean is used as a criteria.
36       *
37       * You can create your own criteria or use one of the static
38       * methods present in this class. These methods will create a
39       * pre-defined criteria.
40       *
41       * The {@link Closure} used as a criteria will receive a {@link
42       * MethodNode} and return a {@link Boolean} value. If the closure
43       * returns true then the transformation will be apply, if not the
44       * {@link MethodNode} will be discarded.
45       *
46       * @deprecated
47       * @param sourceUnit Needed to apply scope
48       * @param criteria used to locate target classes
49       * look for the qualified method)
50       * @since 0.2.0
51       * @see asteroid.Criterias#byMethodNodeNameContains(String)
52       */
53      public AbstractMethodNodeTransformer(final SourceUnit sourceUnit, final Closure<Boolean> criteria) {
54          super(sourceUnit);
55          this.criteria = criteria;
56      }
57  
58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public void visitMethod(final MethodNode methodNode) {
63          if (methodNode == null || !this.criteria.call(methodNode)) {
64              return;
65          }
66  
67          transformMethod(methodNode);
68      }
69  
70      /**
71       * Within this method developer could modify the {@link MethodNode}
72       * instance.
73       *
74       * @param methodNode the {@link MethodNode}  you want to transform
75       * @since 0.2.0
76       */
77      public abstract void transformMethod(MethodNode methodNode);
78  
79  }
