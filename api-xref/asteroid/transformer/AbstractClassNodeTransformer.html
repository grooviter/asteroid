1   package asteroid.transformer;
2   
3   import org.codehaus.groovy.ast.ClassNode;
4   import org.codehaus.groovy.control.SourceUnit;
5   import groovy.lang.Closure;
6   
7   /**
8    * This {@link Transformer} can be used to transform {@link ClassNode}
9    * nodes. To locate the classes you want to transform, a criteria in
10   * form of {@link Closure} can be passed in the constructor.
11   * The criteria must be a {@link Closure} receiving as unique parameter
12   * a {@link ClassNode} and returning a {@link Boolean} value. If the closure
13   * returns true, then the transformation will be applied.
14   * </br></br>
15   * There are some static methods creating some default criterias:
16   * </br></br>
17   * <ul>
18   *    <li>{@link asteroid.Criterias#byClassNodeNameContains(String)}</li>
19   *    <li>{@link asteroid.Criterias#byClassNodeNameStartsWith(String)}</li>
20   *    <li>{@link asteroid.Criterias#byClassNodeNameEndsWith(String)}</li>
21   * </ul>
22   *
23   * @since 0.2.0
24   */
25  public abstract class AbstractClassNodeTransformer extends AbstractTransformer {
26  
27      private final Closure<Boolean> criteria;
28  
29      /**
30       * Every instance needs the source unit and the name of the class
31       * it's going to transform.
32       *
33       * To find classes eligible to be transformed by this {@link Transformer}
34       * a {@link Closure} returning a boolean is used as a criteria.
35       *
36       * You can create your own criteria or use one of the static
37       * methods present in this class. These methods will create a
38       * pre-defined criteria.
39       *
40       * The {@link Closure} used as a criteria will receive a {@link
41       * ClassNode} and return a {@link Boolean} value. If the closure
42       * returns true then the transformation will be apply, if not the
43       * {@link ClassNode} will be discarded.
44       *
45       * @param sourceUnit Needed to apply scope
46       * @param criteria used to locate target classes
47       * look for the qualified class)
48       * @since 0.2.0
49       * @see asteroid.Criterias#byClassNodeNameContains(String)
50       */
51      public AbstractClassNodeTransformer(final SourceUnit sourceUnit, final Closure<Boolean> criteria) {
52          super(sourceUnit);
53          this.criteria = criteria;
54      }
55  
56      /**
57       * {@inheritDoc}
58       */
59      @Override
60      public void visitClass(final ClassNode classNode) {
61          if (classNode == null || !this.criteria.call(classNode)) {
62              return;
63          }
64  
65          transformClass(classNode);
66      }
67  
68      /**
69       * Within this method developer could modify the {@link ClassNode}
70       * instance.
71       *
72       * @param classNode the {@link ClassNode}  you want to transform
73       * @since 0.2.0
74       */
75      public abstract void transformClass(final ClassNode classNode);
76  
77  }
