1   package asteroid.transformer;
2   
3   import groovy.lang.Closure;
4   import org.codehaus.groovy.control.SourceUnit;
5   import org.codehaus.groovy.ast.expr.Expression;
6   
7   /**
8    * This {@link Transformer} could be used as a base for transforming
9    * {@link Expression} instances.
10   * <br><br>
11   * <b class="warning">IMPORTANT:</b> The parameter type is only used as a hint. If you
12   * are not carefull on how you define the search criteria you could
13   * get a {@link ClassCastException} at runtime. The criteria should ask
14   * for the type of the expression in the first place.
15   *
16   * @param <T> use as a hint for the {@link AbstractExpressionTransformer#transformExpression} method parameter
17   * @since 0.2.0
18   */
19  public abstract class AbstractExpressionTransformer<T extends Expression> extends AbstractTransformer {
20  
21      private final Closure<Boolean> criteria;
22      private final Class<T> clazz;
23  
24      /**
25       * Every instance needs the source unit awareness and the name of the method
26       * it's going to transform
27       *
28       * @param clazz the type of the expression we're interested in
29       * @param sourceUnit Needed to apply scope
30       * @param criteria the criteria used to search the interesting
31       * {@link Expression}
32       * @since 0.2.3
33       */
34      public AbstractExpressionTransformer(final Class<T> clazz, final SourceUnit sourceUnit, final Closure<Boolean> criteria) {
35          super(sourceUnit);
36          this.clazz = clazz;
37          this.criteria = criteria;
38      }
39  
40      /**
41       * Every instance needs the source unit awareness and the name of the method
42       * it's going to transform
43       *
44       * @param clazz the type of the expression we're interested in
45       * @param sourceUnit Needed to apply scope
46       * {@link Expression}
47       * @since 0.2.3
48       */
49      public AbstractExpressionTransformer(final Class<T> clazz, final SourceUnit sourceUnit) {
50          super(sourceUnit);
51          this.clazz = clazz;
52          this.criteria = everything();
53      }
54  
55      /**
56       * {@inheritDoc}
57       */
58      @Override
59      public Expression transform(final Expression expression) {
60          if (expression == null) {
61              return null;
62          }
63  
64          final boolean matches = isOfTypeAndMatchesCriteria(expression, clazz, criteria);
65  
66          if (matches) {
67              return this.transformExpression((T) expression);
68          }
69  
70          return expression.transformExpression(this);
71      }
72  
73      private static boolean isOfTypeAndMatchesCriteria(final Expression expression, final Class clazz, final Closure<Boolean> predicate) {
74          return isOfType(expression, clazz) && predicate.call(expression);
75      }
76  
77      private static boolean isOfType(final Expression expression, final Class clazz) {
78          return clazz != null && clazz.isInstance(expression);
79      }
80  
81      /**
82       * This method will transform the expression into its final version.
83       *
84       * @param expression the method expression you want to transform
85       * @return the final version of the method expression
86       * @since 0.2.0
87       */
88      public abstract Expression transformExpression(T expression);
89      /**
90       * This criteria will make the transformer to process every {@link Expression}
91       *
92       * @deprecated use {@link asteroid.Criterias}
93       * @return a criteria to process everything
94       * @since 0.2.0
95       */
96      @Deprecated
97      public static Closure<Boolean> everything() {
98          return new Closure<Boolean>(null) {
99              public Boolean doCall(final Expression expression) {
100                 return true;
101             }
102         };
103     }
104 }
