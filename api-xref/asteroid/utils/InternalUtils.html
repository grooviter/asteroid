1   package asteroid.utils;
2   
3   import org.codehaus.groovy.runtime.DefaultGroovyMethods;
4   
5   import java.util.List;
6   import java.util.ArrayList;
7   import java.util.Collection;
8   import groovy.lang.Closure;
9   
10  /**
11   * Internal utils to deal with Java code
12   *
13   * @since 0.4.3
14   */
15  public final class InternalUtils {
16  
17      private InternalUtils() {
18          // can't be instantiated
19      }
20  
21      /**
22       * Deals with deprecation of some form of collect, and possible
23       * NPE
24       *
25       * @param iterable the object we'd like to iterate over
26       * @param mapping {@link Closure} to transform items in the iterable object
27       * @return an instance of type {@link List} with transformed items
28       * @since 0.4.3
29       */
30      public static <T> List<T> collect(final Object iterable, final Closure<T> process) {
31          if (iterable == null) {
32              return new ArrayList<>();
33          }
34  
35          return DefaultGroovyMethods.collect(iterable, process);
36      }
37  
38      /**
39       * Deals with deprecation of some form of collect, and possible
40       * NPE
41       *
42       * @param iterable the object we'd like to iterate over
43       * @return an instance of type {@link List}
44       * @since 0.4.3
45       */
46      public static <T> List<T> collect(final Collection<T> iterable) {
47          if (iterable == null) {
48              return new ArrayList<>();
49          }
50  
51          return new ArrayList<>(DefaultGroovyMethods.collect(iterable.iterator()));
52      }
53  }
